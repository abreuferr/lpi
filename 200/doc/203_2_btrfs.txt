# BTRFS
#
# - forte sistema contra possivel falha e erros;
# - possibilidade de desfragmentar e redimensionar o disco com
# ele montado;
# - snapshot com a particao montada;
# - raid 0/1/10;
# - sistema de arquivos do tipo copy-on-write.

# instalacao do software
#
$ sudo apt-get install btrfs-tools

#
# FORMATACAO
#

# formatacao do sistema de arquivos do tipo BTRFS
#
# SEM OPCAO
#
# nesse tipo de formatacao, os dados sao armazenados
# como um RAID0 e os metadados como um RAID1.
#
$ sudo mkfs -t btrfs /dev/sd[bc]
	Label:              (null)
	UUID:
	Node size:          16384
	Sector size:        4096
	Filesystem size:    40.00GiB
	Block group profiles:
	  Data:             RAID0             2.00GiB
	  Metadata:         RAID1             1.00GiB
	  System:           RAID1             8.00MiB
	SSD detected:       no
	Incompat features:  extref, skinny-metadata
	Number of devices:  2
	Devices:
	   ID        SIZE  PATH
	    1    20.00GiB  /dev/sdb
	    2    20.00GiB  /dev/sdc

# formatacao do sistema de arquivos do tipo BTRFS
#
# SINGLE
#
# -m - metadados
#
$ sudo mkfs -t btrfs -m single /dev/sd[bc]

# formatacao do sistema de arquivos do tipo BTRFS
#
# RAID 0
#
# -m - metadados
# -d - dados
#
$ sudo mkfs -t btrfs -m raid0 /dev/sd[bc]

# formatacao do sistema de arquivos do tipo BTRFS
#
# RAID 1
#
# -m - metadados (RAID 1 - default)
# -d - dados
#
$ sudo mkfs -t btrfs -d raid1 /dev/sd[bc]

# formatacao do sistema de arquivos do tipo BTRFS
#
# RAID 10
#
# -m - metadados
# -d - dados
#
$ sudo mkfs -t btrfs -d raid1 -m raid0 /dev/sd[bc]

	Filesystem      Size  Used Avail Use% Mounted on
	/dev/sdb         60G   17M   59G   1% /mnt/btrfs

#
# SUBVOLUME
#

# criar subvolume
#
$ sudo btrfs subvolume create /mnt/btrfs/subvol

# listar subvolume
#
$ sudo btrfs subvolume list /mnt/btrfs/

# apagar subvolume
#
$ sudo btrfs subvolume delete /mnt/btrfs/subvol

# montar subvolume
#
$ sudo btrfs subvolume list /mnt/btrfs/
	ID 257 gen 13 top level 5 path subvol
$ sudo umount /mnt/btrfs/
$ sudo mkdir /mnt/subvol
$ sudo mount -o subvolid=257 /dev/sdb /mnt/subvol/

#
# SNAPSHOT
#

# criar snapshot
#
$ sudo btrfs subvolume snapshot /mnt/btrfs/subvol/ /mnt/btrfs/subvol_snap/

# snapshot em modo read-only
#
$ sudo btrfs subvolume snapshot -r /mnt/subvol/ /mnt/subvol_snap_r/
